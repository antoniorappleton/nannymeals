function calcMatch(meal, pantry) {
const needed = meal.pantryKeys || [];
if (!needed.length) return null;
const have = needed.filter(n => { const p = pantry.find(x=>x.name===n); return p && p.status!=="return Math.round((have.length / needed.length) * 100);
}
function fillWeek(profile, pantry) {
const budget = parseFloat((profile.budget||"50-80").split("-")[0]) || 50;
const tight = budget < 60;
const scored = ALL_MEALS.map(m => {
let score = (calcMatch(m, pantry)||0) * 0.3;
if (tight) score += (10 - m.costVal) * 4;
score += Math.random() * 10;
return { ...m, score };
}).sort((a,b) => b.score - a.score);
const selections = {};
const usedIds = new Set();
const tagOrder = ["Fish","Meat","Veggie","Meat","Veggie","Fish","Veggie"];
DAYS.forEach((d,i) => {
const pool = scored.filter(m => !usedIds.has(m.id) && m.tag===tagOrder[i]);
const pick = (pool.length ? pool : scored.filter(m=>!usedIds.has(m.id)))[0];
if (pick) { selections[d] = pick; usedIds.add(pick.id); }
});
return selections;
}
